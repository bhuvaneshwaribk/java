        }

 

      }

    }

  }

41. Write a JAVAprogram to find sum of all prime numbers between 1 to n.
import java.util.Scanner;

public class PrimeNumber {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter a number (n): ");
        int n = scanner.nextInt();

        int sum = 0;

        for (int i = 2; i <= n; i++) {
            if (isPrime(i)) {
                sum += i;
            }
        }

        System.out.println("Sum of prime numbers between 1 and " + n + " is: " + sum);

        scanner.close();
    }

    // Function to check if a number is prime
    public static boolean isPrime(int num) {
        if (num <= 1) {
            return false;
        }
        if (num <= 3) {
            return true;
        }
        if (num % 2 == 0 || num % 3 == 0) {
            return false;
        }
        for (int i = 5; i * i <= num; i += 6) {
            if (num % i == 0 || num % (i + 2) == 0) {
                return false;
            }
        }
        return true;
    }
}

public class PalindromeCheck {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String str = scanner.nextLine();

        if (isPalindrome(str)) {
            System.out.println("The string is a palindrome.");
        } else {
            System.out.println("The string is not a palindrome.");
        }
    }

    public static boolean isPalindrome(String str) {
        int left = 0;
        int right = str.length() - 1;

        while (left < right) {
            if (str.charAt(left) != str.charAt(right)) {
                return false;
            }
            left++;
            right--;
        }

        return true;
    }
}

        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == target) {
                count++;
            }

        int currentIndex = 0;

        for (int i = 0; i < str.length(); i++) {
            if (str.charAt(i) == target) {
                occurrences[currentIndex] = i;
                currentIndex++;
            }
        }

        return occurrences;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String input = scanner.nextLine();
        scanner.close();

        char highestFrequencyChar = findHighestFrequencyCharacter(input);
        
        if (highestFrequencyChar != '\0') {
            System.out.println("The character '" + highestFrequencyChar + "' has the highest frequency.");
        } else {
            System.out.println("The input string is empty.");
        }
    }

    public static char findHighestFrequencyCharacter(String str) {
        if (str.isEmpty()) {
            return '\0'; // Return null character if the string is empty
        }

        Map<Character, Integer> frequencyMap = new HashMap<>();
        
        for (char ch : str.toCharArray()) {
            frequencyMap.put(ch, frequencyMap.getOrDefault(ch, 0) + 1);
        }
        
        char highestFrequencyChar = '\0'; // Default value if there are no characters
        int maxFrequency = 0;
        
        for (Map.Entry<Character, Integer> entry : frequencyMap.entrySet()) {
            if (entry.getValue() > maxFrequency) {
                maxFrequency = entry.getValue();
                highestFrequencyChar = entry.getKey();
            }
        }
        
        return highestFrequencyChar;
    }
}
